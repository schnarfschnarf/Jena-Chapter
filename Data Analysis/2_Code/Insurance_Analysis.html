#load synthesis dataset
synthDat <- read_xlsx(
  here("1_Data", "economic_MF_final.xlsx")
) %>%
  filter(sowndiv < 60) %>%
  mutate(block = as.factor(block))

synthDat$plotcode <- as.factor(synthDat$plotcode) #change the plotcode to a factor
# define model for RP ~ sowndiv
fit_rp <- fit_or_load("fit_rp", function() {
  brm(
  RP ~ log2(sowndiv) + (1|block),
  data = synthDat
)
})

# plot regression
c_sown_rp <- conditional_effects(fit_rp)
plot_sown_rp_data <- c_sown_rp[[1]]

p_sown_rp <- ggplot() +
  geom_point(
    data   = synthDat,
    aes(x  = sowndiv, y = RP),
    alpha  = 0.5,
    size   = 2,
    shape  = 21,
    fill   = "#0072B2",
    colour = "#004C70",
    stroke = 0.4,
    position = position_jitter(width = 0.12, height = 0)
  ) +
  geom_line(
    data     = plot_sown_rp_data,
    aes(x    = 2^log2(sowndiv), y = estimate__),
    colour   = "#0072B2",
    linewidth = 0.8
  ) +
  geom_ribbon(
    data  = plot_sown_rp_data,
    aes(x = 2^log2(sowndiv), ymin = lower__, ymax = upper__),
    fill  = "#0072B2",
    alpha = 0.15
  ) +
  scale_x_continuous(
    breaks = c(1, 2, 4, 8, 16),
    labels = c(1, 2, 4, 8, 16),
    expand = expansion(mult = c(0.02, 0.02))
  ) +
  labs(
    x = "Sown plant species richness",
    y = expression("Risk premium; € ha"^{-1}*" yr"^{-1}*"")
  ) +
  coord_cartesian(clip = "off") +
  theme_custom() +
  theme(aspect.ratio = 0.6)

# calculate effect size of insurance
post <- as_draws_df(fit_rp)

insurance_draws <- -post$b_log2sowndiv

insurance_summary <- tibble(
  mean = mean(insurance_draws),
  lwr  = quantile(insurance_draws, 0.025),
  upr  = quantile(insurance_draws, 0.975)
)
