# Function to fill gaps using linear regression
fill_gaps_with_regression <- function(data) {
  plots <- unique(data$plot)
  
  for (plot in plots) {
    plot_data <- data %>% filter(plot == !!plot)
    
    # Fit linear model if there are enough non-NA values
    if (sum(!is.na(plot_data$NF1)) > 1) {
      lm_model <- lm(NF1 ~ NS_Summe_mm, data = plot_data, na.action = na.exclude)
      missing_indices <- which(is.na(plot_data$NF1))
      
      # Predict missing values
      if (length(missing_indices) > 0) {
        plot_data$NF1[missing_indices] <- predict(lm_model, newdata = plot_data[missing_indices, ])
      }
      
      # Update the original data frame
      data[data$plot == plot, ] <- plot_data
    }
  }
  
  return(data)
}

# Apply the function to your data frame
leach_x1_filled <- fill_gaps_with_regression(leach_x1)

# Function to fill gaps with the average values per year using a for loop
fill_gaps_with_yearly_avg <- function(data) {
  data %>%
    group_by(plot, year) %>%
    mutate(yearly_avg = ifelse(all(is.na(tn)), NA, mean(tn, na.rm = TRUE))) %>%
    mutate(tn = ifelse(is.na(tn), yearly_avg, tn)) %>%
    select(-yearly_avg) %>%
    ungroup()
}

# Apply the function to your data frame
